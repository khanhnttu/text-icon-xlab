```bash
npm install --save text-icon-xlab
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'text-icon-xlab'
import 'text-icon-xlab/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
